#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main() {
    srand(time(NULL));
    int campo[5][5];
    int x,y,z;
    char risultato[5][5];
    for(int i=0;i<5;i++)
    {
        for(int j=0;j<5;j++)
        {
            campo[i][j]=0;
            
        }
    }
    for(int i=0;i<3;i++)
    {
        x=rand() % 5;
        y=rand() % 5;
        if(campo[x][y] == 1)
        {
            i--;
        }
        else
        {
            campo[x][y] = 1;
        }
        
    }
    printf("il giochino inizia:)\n");
    for(int i=0;i<=7;i++)
    {
        
        if(i==7)
        {
            printf("Hai perso :(\n");
            break;
        }
        printf("Inserisci la riga dove sparare\n");
        scanf("%d",&x);
        printf("Inserisci la colonna dove sparare\n");
        scanf("%d",&y);
        if(campo[x][y] == 2)
        {
            printf("nave gia' colpita\n");
            i--;
        }
        if(campo[x][y] == 1)
        {
            printf("COLPITO!\n");
            campo[x][y] = 2;
            z++;
        }
        else if(campo[x][y] == 0)
        {
            printf("ACQUA!\n");
            campo[x][y] = 3;
        }
        if(z==3)
        {
            printf("HAI VINTO!!\n");
            break;
        }
          
    }
    for(int i=0;i<5;i++)
    {
        for(int j=0;j<5;j++)
        {
            if(campo[i][j]==2)
            {
                risultato[i][j]='X';
            }
            else if(campo[i][j]==0)
            {
                risultato[i][j]='~';
            }
            else if(campo[i][j]==1)
            {
                risultato[i][j]='N';
            }
            else
            {
                risultato[i][j]='O';
            }
            
        }
    }
    for(int i=0;i<5;i++)
    {
        for(int j=0;j<5;j++)
        {
            printf("%c",risultato[i][j]);
            
        }
        printf("\n");
    }
    return 0;
}
